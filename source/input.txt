event sec = timerSec(int a){
	print(a)
}


state aa{
	entry(){
		init sec(3)
		init timerSec(4)
	}
	timerSec(int n){
		print(n)
	}
	sec()
	timerSec(int c){
		print(c)
		if(c>2){
			state cc
		}
	}
}
